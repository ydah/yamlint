def alias(anchor)
  mark = @parser&.mark
  @aliases << {
    name: anchor,
    line: mark ? mark.line + 1 : 1,
    column: mark ? mark.column + 1 : 1
  }
end

def unused_anchors
  used_names = @aliases.map { |a| a[:name] }
  @anchors.except(*used_names).map do |_, info|
    info.is_a?(Array) ? info.first : info
  end
end
