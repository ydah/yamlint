def check_line(line, line_number, _context)
  return if line.chomp.empty?

  trailing = line[/[ \t]+$/]
  return unless trailing

  problem(
    line: line_number,
    column: line.rstrip.length + 1,
    message: 'trailing spaces',
    fixable: true
  )
end
