def self.get(format)
  case format.to_s
  when 'standard'
    Standard.new
  when 'parsable'
    Parsable.new
  when 'colored', 'auto'
    Colored.new
  when 'github'
    Github.new
  else
    raise ArgumentError, "Unknown output format: #{format}"
  end
end
