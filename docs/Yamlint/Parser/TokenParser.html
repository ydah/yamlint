def parse
  handler = Handler.new
  parser = Psych::Parser.new(handler)
  handler.parser = parser

  begin
    parser.parse(@content)
  rescue Psych::SyntaxError
    # Syntax errors will be handled separately
  end

  handler.tokens
end
